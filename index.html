Quick Start

npm 
npm install @agentport/core
npm install @agentport/eliza  # Optional Eliza agent
npm install @agentport/rig    # Optional Rig agent
index.ts 
import { AgentPort } from '@agentport/core'
import { ElizaAgent } from '@agentport/eliza'
import { RigAgent } from '@agentport/rig'

// Initialize AgentPort
const port = new AgentPort()

// Add built-in agents
port.use(new ElizaAgent())
port.use(new RigAgent())

// Add your custom agent
port.use({
  name: 'custom',
  async process(ctx) {
    return 'Hello from custom agent!'
  }
})

// Start the server
port.listen(3000)
