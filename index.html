Quick Start

Type-safe AI agent orchestration with TypeScript

// Install AgentPort
npm install agentport

// Import types and AgentPort class
import {
    AgentPort,
    type AgentConfig,
    type AgentResponse
} from 'agentport'

// Define agent configuration type
interface LLMConfig extends AgentConfig {
    type: 'llm'
    model: 'gpt-4' | 'gpt-3.5-turbo'
    config: {
        temperature: number
        maxTokens?: number
    }
}

// Initialize and connect an agent
const port = new AgentPort()
const agent = await port.connect<LLMConfig>({
    type: 'llm',
    model: 'gpt-4',
    config: {
        temperature: 0.7,
        maxTokens: 2048
    }
})

// Define task parameters type
interface AnalyzeTaskParams {
    task: 'analyze_text'
    input: string
}

// Execute task with type safety
const result = await agent.execute<AnalyzeTaskParams, AgentResponse>({
    task: 'analyze_text',
    input: 'Your text here'
})
